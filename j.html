import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

public class GreetingApp {

 JFrame frame = new JFrame("Time-Based Greetings");
 frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
 frame.setSize(300, 150);
 frame.setLayout(null);
 JButton greetButton = new JButton("Greet");
 greetButton.setBounds(100, 40, 100, 30);
 frame.add(greetButton);
 JLabel greetingLabel = new JLabel();
 greetingLabel.setBounds(100, 80, 200, 30);
 frame.add(greetingLabel);
 greetButton.addActionListener(new ActionListener() {
 @Override
 public void actionPerformed(ActionEvent e) {
 Calendar calendar = Calendar.getInstance();
 int hour = calendar.get(Calendar.HOUR_OF_DAY);
 String greeting;
 if (hour >= 0 && hour < 12) {
 greeting = "Good Morning";
 } else if (hour >= 12 && hour < 17) {
 greeting = "Good Afternoon";
 } else {
 greeting = "Good Evening";
 }
 greetingLabel.setText(greeting);
 }
 });
 frame.setVisible(true);
 }
}

import javax.swing.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
public class SimpleCalculator {
 public static void main(String[] args) {
 JFrame frame = new JFrame("Simple Calculator");
 frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
 frame.setSize(300, 150);
 frame.setLayout(new java.awt.GridLayout(4, 2));
 JTextField num1Field = new JTextField(10);
 JTextField num2Field = new JTextField(10);
 JTextField resultField = new JTextField(10);
 resultField.setEditable(false);
 frame.add(new JLabel("Number 1:"));
 frame.add(num1Field);
 frame.add(new JLabel("Number 2:"));
 frame.add(num2Field);
 frame.add(new JLabel("Result:"));
 frame.add(resultField);
 String[] operators = {"Add", "Subtract", "Multiply", "Divide"};
 JButton[] operationButtons = new JButton[operators.length];
 for (int i = 0; i < operators.length; i++) {
 operationButtons[i] = new JButton(operators[i]);
 frame.add(operationButtons[i]);
 }
 ActionListener operatorListener = new ActionListener() {
 @Override
 public void actionPerformed(ActionEvent e) {
 double num1 = Double.parseDouble(num1Field.getText());
 double num2 = Double.parseDouble(num2Field.getText());
 double result = 0;
 String operator = e.getActionCommand();
 switch (operator) {
 case "Add":
 result = num1 + num2;
break;
 case "Subtract":
 result = num1 - num2;
 break;
 case "Multiply":
 result = num1 * num2;
break;
 case "Divide":
 if (num2 != 0) {
 result = num1 / num2;
 } else {
 JOptionPane.showMessageDialog(frame, "Cannot
divide by zero.");
 return;
 }
break;
 }
 resultField.setText(Double.toString(result));
 }
 };
 for (JButton button : operationButtons) {
 button.addActionListener(operatorListener);
 }
 frame.setVisible(true);
 }
}

FileNameExtensionFilter("Text Files", "txt");
 fileChooser.setFileFilter(filter);
 int returnValue = fileChooser.showOpenDialog(frame);
 if (returnValue == JFileChooser.APPROVE_OPTION) {
 File selectedFile = fileChooser.getSelectedFile();
try (BufferedReader reader = new BufferedReader(new
FileReader(selectedFile))) {
 StringBuilder fileContent = new StringBuilder();
String line;
while ((line = reader.readLine()) != null) {
 fileContent.append(line).append("\n");
 }
textArea.setText(fileContent.toString());

 ex.printStackTrace();
JOptionPane.showMessageDialog(frame, "Error
opening the file", "Error", JOptionPane.ERROR_MESSAGE);
 }

 }

 return id;

 return name;

import java.util.Arrays;
import java.util.Comparator;
public class PairSorter {
 public static void main(String[] args) {
 Pair[] pairs = {
 new Pair("apple", 5),
 new Pair("banana", 2),
 new Pair("apple", 2),
 new Pair("banana", 5)
 };
 Arrays.sort(pairs, new Comparator() {
 @Override
 public int compare(Pair pair1, Pair pair2) {
 int stringComparison =
pair1.getKey().compareTo(pair2.getKey());
 if (stringComparison != 0) {
 return stringComparison;
 } else {
 return Integer.compare(pair1.getValue(),
pair2.getValue());
 }
 }
 });
 for (Pair pair : pairs) {
 System.out.println(pair.getKey() + ", " + pair.getValue());
 }
 }

 private int value;

 this.value = value;

 public String getKey() {
 return key;

 return value;

import java.util.ArrayList;
import java.util.Collections;
import java.util.Comparator;
class Student {
 private String name;
 private int age;
 public Student(String name, int age) {
 this.name = name;
 this.age = age;
 }
 public String getName() {
 return name;
 }
 public int getAge() {
 return age;
 }
}
public class StudentSorting {
 public static void main(String[] args) {
 ArrayList students = new ArrayList<>();
 students.add(new Student("Alice", 20));
 students.add(new Student("Bob", 22));
 students.add(new Student("Alice", 19));
 students.add(new Student("Bob", 25));
 // Create a custom comparator
 Comparator comparator = Comparator
 .comparing(Student::getName)
 .thenComparingInt(Student::getAge);
 // Sort the ArrayList using the custom comparator
 Collections.sort(students, comparator);
 // Display the sorted list
 for (Student student : students) {

student.getAge());
 }
 }
